# Higher-Order Functions

---

# 🔹 1. What is a Higher-Order Function?

A function that **receives another function as an argument**, or **returns a function**, or **both**.

```
Normal Function   → takes values as input
Higher-Order Function → takes functions as input OR returns a function
```

```javascript
// Normal function passed as argument
function greet() {
    console.log("Hello!");
}

function run(fn) {  // ← receives a function
    fn();
}

run(greet); // Hello!
```

---

# 🔹 2. Types of Higher-Order Functions

```
1. Function that accepts a function as argument
2. Function that returns a function
3. Function that does both
```

---

# 🔹 3. Type 1 — Accepts a Function as Argument

---

### ➤ Basic Example

```javascript
function doMath(a, b, operation) {
    return operation(a, b);
}

function add(a, b) {
    return a + b;
}

function multiply(a, b) {
    return a * b;
}

console.log(doMath(5, 3, add));      // 8
console.log(doMath(5, 3, multiply)); // 15
```

---

### ➤ With Arrow Functions

```javascript
function doMath(a, b, operation) {
    return operation(a, b);
}

console.log(doMath(5, 3, (a, b) => a + b)); // 8
console.log(doMath(5, 3, (a, b) => a * b)); // 15
```

---

### ➤ Real Example — Run After Delay

```javascript
function runAfterTask(task, callback) {
    task();
    callback();
}

function mainTask() {
    console.log("Main task done");
}

function onComplete() {
    console.log("Now running callback");
}

runAfterTask(mainTask, onComplete);
// Main task done
// Now running callback
```

---

### ➤ Real Example — Filter Users by Condition

```javascript
function filterUsers(users, condition) {
    let result = [];
    for (let user of users) {
        if (condition(user)) {
            result.push(user);
        }
    }
    return result;
}

let users = [
    { name: "Ali",    age: 17 },
    { name: "Kashif", age: 22 },
    { name: "Sara",   age: 15 },
    { name: "Ahmed",  age: 25 }
];

let adults = filterUsers(users, user => user.age >= 18);

console.log(adults);
// [{ name: "Kashif", age: 22 }, { name: "Ahmed", age: 25 }]
```

# 🔹 4. Type 2 — Returns a Function

---

### ➤ Basic Example

```javascript
function createGreeter(greeting) {
    return function(name) {
        console.log(greeting + ", " + name);
    };
}

let sayHello = createGreeter("Hello");
let sayHi    = createGreeter("Hi");

sayHello("Kashif"); // Hello, Kashif
sayHi("Ali");       // Hi, Ali
```

---

### ➤ Same with Arrow Functions

```javascript
const createGreeter = (greeting) => (name) => {
    console.log(greeting + ", " + name);
};

let sayHello = createGreeter("Hello");

sayHello("Kashif"); // Hello, Kashif
```

---

### ➤ Real Example — Multiplier Factory

```javascript
function createMultiplier(factor) {
    return function(number) {
        return number * factor;
    };
}

let double = createMultiplier(2);
let triple = createMultiplier(3);

console.log(double(5));  // 10
console.log(triple(5));  // 15
console.log(double(10)); // 20
```

---

### ➤ Real Example — Role-Based Access

```javascript
function createAccessChecker(requiredRole) {
    return function(user) {
        if (user.role === requiredRole) {
            console.log(user.name + " — Access granted");
        } else {
            console.log(user.name + " — Access denied");
        }
    };
}

let checkAdmin  = createAccessChecker("admin");
let checkEditor = createAccessChecker("editor");

checkAdmin({ name: "Kashif", role: "admin" });  // Kashif — Access granted
checkAdmin({ name: "Ali",    role: "viewer" }); // Ali    — Access denied
checkEditor({ name: "Sara",  role: "editor" }); // Sara   — Access granted
```

# 🔹 5. Type 3 — Accepts and Returns a Function

---

### ➤ Basic Example

```javascript
function applyTwice(fn) {
    return function(value) {
        return fn(fn(value));
    };
}

const addThree = n => n + 3;

let addSix = applyTwice(addThree);

console.log(addSix(10)); // 16  → addThree(addThree(10)) → addThree(13) → 16
```

---

### ➤ Real Example — Logger Wrapper

```javascript
function withLogger(fn) {
    return function(...args) {
        console.log("Calling with:", args);
        let result = fn(...args);
        console.log("Result:", result);
        return result;
    };
}

const add = (a, b) => a + b;

let loggedAdd = withLogger(add);

loggedAdd(3, 5);
// Calling with: [3, 5]
// Result: 8
```

# 🔹 6. Built-in Higher-Order Functions

JavaScript arrays come with built-in higher-order functions.

---

### ➤ `map()` — Transform every element

```javascript
let numbers = [1, 2, 3, 4];

let squared = numbers.map(n => n * n);

console.log(squared); // [1, 4, 9, 16]
```

---

### ➤ `filter()` — Keep elements that pass condition

```javascript
let numbers = [1, 2, 3, 4, 5, 6];

let odds = numbers.filter(n => n % 2 !== 0);

console.log(odds); // [1, 3, 5]
```

---

### ➤ `reduce()` — Reduce to single value

```javascript
let numbers = [1, 2, 3, 4, 5];

let total = numbers.reduce((acc, curr) => acc + curr, 0);

console.log(total); // 15
```

---

### ➤ `forEach()` — Loop with no return

```javascript
let names = ["Ali", "Kashif", "Sara"];

names.forEach(name => console.log("Hello, " + name));
// Hello, Ali
// Hello, Kashif
// Hello, Sara
```

---

### ➤ `sort()` — Sort with custom comparator

```javascript
let users = [
    { name: "Zara",   age: 25 },
    { name: "Ali",    age: 17 },
    { name: "Kashif", age: 22 }
];

users.sort((a, b) => a.age - b.age); // ascending by age

console.log(users);
// { name: "Ali", age: 17 }
// { name: "Kashif", age: 22 }
// { name: "Zara", age: 25 }
```

---

### ➤ Chaining Built-in HOFs

```javascript
let orders = [
    { item: "Laptop", price: 1200, delivered: true  },
    { item: "Phone",  price: 800,  delivered: false },
    { item: "Watch",  price: 200,  delivered: true  },
    { item: "Tablet", price: 600,  delivered: false }
];

let total = orders
    .filter(order => order.delivered)        // keep delivered only
    .map(order => order.price)               // extract prices
    .reduce((acc, price) => acc + price, 0); // sum them

console.log(total); // 1400
```

# 🔹 7. Callback Function

A function **passed as an argument** to another function is called a **callback**.

```javascript
function greet(name, callback) {
    console.log("Hello, " + name);
    callback(); // runs after greet
}

function afterGreet() {
    console.log("How are you?");
}

greet("Kashif", afterGreet);
// Hello, Kashif
// How are you?
```

---

### ➤ Inline Callback

```javascript
greet("Kashif", () => {
    console.log("How are you?");
});
// Hello, Kashif
// How are you?
```

---

# 🔥 Final Summary Table

| Type | What it does | Example |
|------|-------------|---------|
| Accepts function | Takes a function as argument | `filter(fn)`, `map(fn)` |
| Returns function | Creates and gives back a function | `createMultiplier(2)` |
| Both | Takes and returns a function | `withLogger(fn)` |
| Callback | Function passed to be called later | `greet("Ali", fn)` |

---

# 💡 Key Takeaway

- **Higher-order functions** make code reusable and flexible
- **Type 1** → pass behavior as argument (what to do is decided outside)
- **Type 2** → factory pattern, create specialized functions dynamically
- **Type 3** → wrap or enhance existing functions
- **Built-in HOFs** like `map`, `filter`, `reduce` are the most common use in daily code
- A function passed as argument is called a **callback**
