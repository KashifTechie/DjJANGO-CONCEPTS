## 👤 Step 3 — Custom User Model

> ⚠️ **Do this before your first `migrate`.** Changing `AUTH_USER_MODEL` after migrations exist is a nightmare.

```python
# users/models.py

from django.contrib.auth.models import AbstractUser
from django.db import models


class User(AbstractUser):
    email = models.EmailField(unique=True)
    phone_number = models.CharField(max_length=15, blank=True, null=True)
    is_verified = models.BooleanField(default=False)  # True after OTP verified

    USERNAME_FIELD = "email"        # login with email, not username
    REQUIRED_FIELDS = ["username"]  # still required for createsuperuser

    def __str__(self):
        return self.email
```

Now run migrations:

```bash
python manage.py makemigrations
python manage.py migrate
```

## 🔢 Step 4 — OTP Utility (Redis)

OTPs are temporary — Redis is perfect for this. They auto-expire, no cleanup needed.

```python
# users/otp_utils.py

import random
from django.core.cache import cache

PREFIX = "otp:"
OTP_TTL = 300   # 5 minutes


def generate_otp():
    return str(random.randint(100000, 999999))


def store_otp(email, otp):
    cache.set(f"{PREFIX}{email}", otp, timeout=OTP_TTL)


def verify_otp(email, otp):
    stored = cache.get(f"{PREFIX}{email}")
    if stored and stored == otp:
        cache.delete(f"{PREFIX}{email}")    # one-time use, delete immediately
        return True
    return False
```

## 🛡️ Step 5 — Redis Blacklist Utility

Used to invalidate tokens on logout.

```python
# authentication/redis_blacklist.py

from django.core.cache import cache

PREFIX = "jwt_blacklist:"


def blacklist_token(jti, ttl):
    cache.set(f"{PREFIX}{jti}", "1", timeout=ttl)


def is_blacklisted(jti):
    return cache.get(f"{PREFIX}{jti}") is not None
```

## 🔍 Step 6 — Custom JWT Authentication

This intercepts every request and checks if the token has been blacklisted.

```python
# authentication/authentication.py

from rest_framework_simplejwt.authentication import JWTAuthentication
from rest_framework.exceptions import AuthenticationFailed
from .redis_blacklist import is_blacklisted


class CustomJWTAuthentication(JWTAuthentication):

    def get_validated_token(self, raw_token):
        token = super().get_validated_token(raw_token)

        jti = token.get("jti")

        if is_blacklisted(jti):
            raise AuthenticationFailed("Token has been invalidated. Please log in again.")

        return token
```

## 📦 Step 7 — Serializers

```python
# users/serializers.py

from rest_framework import serializers
from django.contrib.auth import get_user_model
from drf_recaptcha.fields import ReCaptchaV2Field

User = get_user_model()


class RegisterSerializer(serializers.ModelSerializer):
    password = serializers.CharField(write_only=True, min_length=8)
    confirm_password = serializers.CharField(write_only=True)
    recaptcha = ReCaptchaV2Field()

    class Meta:
        model = User
        fields = ["username", "email", "password", "confirm_password", "recaptcha"]

    def validate_email(self, value):
        if User.objects.filter(email=value).exists():
            raise serializers.ValidationError("This email is already registered.")
        return value

    def validate(self, data):
        if data["password"] != data["confirm_password"]:
            raise serializers.ValidationError({"confirm_password": "Passwords do not match."})
        return data

    def create(self, validated_data):
        validated_data.pop("recaptcha")
        validated_data.pop("confirm_password")

        user = User.objects.create_user(
            username=validated_data["username"],
            email=validated_data["email"],
            password=validated_data["password"],
            is_verified=False,
            is_active=False,    # blocked from login until OTP verified
        )
        return user


class OTPVerifySerializer(serializers.Serializer):
    email = serializers.EmailField()
    otp = serializers.CharField(min_length=6, max_length=6)


class ResendOTPSerializer(serializers.Serializer):
    email = serializers.EmailField()
```

## 📧 Step 8 — Email Utility

```python
# users/tasks.py

from django.core.mail import send_mail
from django.conf import settings


def send_otp_email(email, otp):
    send_mail(
        subject="Your OTP Verification Code",
        message=f"Your OTP is: {otp}\n\nThis code expires in 5 minutes.\nDo not share it with anyone.",
        from_email=settings.DEFAULT_FROM_EMAIL,
        recipient_list=[email],
        fail_silently=False,
    )
```

> In production, wrap this in a **Celery task** with `@shared_task` so the response isn't blocked waiting for the email to send.

## 🚀 Step 9 — Views

```python
# users/views.py

from rest_framework.views import APIView
from rest_framework.response import Response
from rest_framework import status
from rest_framework.permissions import AllowAny
from django.contrib.auth import get_user_model

from .serializers import RegisterSerializer, OTPVerifySerializer, ResendOTPSerializer
from .otp_utils import generate_otp, store_otp, verify_otp
from .tasks import send_otp_email

User = get_user_model()


# ───────────────────────────────
# 📝 Register
# ───────────────────────────────
class RegisterView(APIView):
    permission_classes = [AllowAny]

    def post(self, request):
        serializer = RegisterSerializer(data=request.data)

        if not serializer.is_valid():
            return Response(serializer.errors, status=status.HTTP_400_BAD_REQUEST)

        user = serializer.save()

        otp = generate_otp()
        store_otp(user.email, otp)
        send_otp_email(user.email, otp)

        return Response(
            {"message": "Account created. Check your email for the OTP."},
            status=status.HTTP_201_CREATED
        )


# ───────────────────────────────
# ✅ Verify OTP
# ───────────────────────────────
class VerifyOTPView(APIView):
    permission_classes = [AllowAny]

    def post(self, request):
        serializer = OTPVerifySerializer(data=request.data)

        if not serializer.is_valid():
            return Response(serializer.errors, status=status.HTTP_400_BAD_REQUEST)

        email = serializer.validated_data["email"]
        otp = serializer.validated_data["otp"]

        if not verify_otp(email, otp):
            return Response(
                {"error": "Invalid or expired OTP."},
                status=status.HTTP_400_BAD_REQUEST
            )

        try:
            user = User.objects.get(email=email)
            user.is_verified = True
            user.is_active = True   # now they can login
            user.save()
        except User.DoesNotExist:
            return Response({"error": "User not found."}, status=status.HTTP_404_NOT_FOUND)

        return Response({"message": "Email verified. You can now log in."})


# ───────────────────────────────
# 🔁 Resend OTP
# ───────────────────────────────
class ResendOTPView(APIView):
    permission_classes = [AllowAny]

    def post(self, request):
        serializer = ResendOTPSerializer(data=request.data)

        if not serializer.is_valid():
            return Response(serializer.errors, status=status.HTTP_400_BAD_REQUEST)

        email = serializer.validated_data["email"]

        try:
            user = User.objects.get(email=email)
        except User.DoesNotExist:
            return Response({"error": "User not found."}, status=status.HTTP_404_NOT_FOUND)

        if user.is_verified:
            return Response({"error": "Account is already verified."}, status=status.HTTP_400_BAD_REQUEST)

        otp = generate_otp()
        store_otp(email, otp)
        send_otp_email(email, otp)

        return Response({"message": "A new OTP has been sent to your email."})
```

```python
# authentication/views.py

import time
from rest_framework.views import APIView
from rest_framework.permissions import IsAuthenticated, AllowAny
from rest_framework.response import Response
from rest_framework import status
from rest_framework_simplejwt.views import TokenObtainPairView
from rest_framework_simplejwt.tokens import RefreshToken

from users.serializers import ReCaptchaSerializer
from .redis_blacklist import blacklist_token


# ───────────────────────────────
# 🔑 Login
# ───────────────────────────────
class LoginView(TokenObtainPairView):
    permission_classes = [AllowAny]

    def post(self, request, *args, **kwargs):
        # Step 1: verify reCAPTCHA
        captcha_serializer = ReCaptchaSerializer(data={"recaptcha": request.data.get("recaptcha")})

        if not captcha_serializer.is_valid():
            return Response({"error": "Bot detected."}, status=status.HTTP_400_BAD_REQUEST)

        # Step 2: JWT login (handles credentials + returns tokens)
        return super().post(request, *args, **kwargs)


# ───────────────────────────────
# 🚪 Logout
# ───────────────────────────────
class LogoutView(APIView):
    permission_classes = [IsAuthenticated]

    def post(self, request):
        refresh_token = request.data.get("refresh")

        if not refresh_token:
            return Response({"error": "Refresh token is required."}, status=status.HTTP_400_BAD_REQUEST)

        try:
            token = RefreshToken(refresh_token)

            jti = token["jti"]
            exp = token["exp"]
            ttl = exp - int(time.time())

            blacklist_token(jti, ttl)

            return Response({"message": "Logged out successfully."})

        except Exception:
            return Response({"error": "Invalid token."}, status=status.HTTP_400_BAD_REQUEST)
```

## 🔁 Complete Request Flow

```
POST /api/auth/register/
  → reCAPTCHA verified
  → user created (is_active=False, is_verified=False)
  → OTP generated → stored in Redis (5 min TTL)
  → OTP sent to email

POST /api/auth/verify-otp/
  → OTP matched from Redis → deleted (one-time use)
  → user.is_active = True
  → user.is_verified = True

POST /api/auth/login/
  → reCAPTCHA verified
  → credentials checked (fails if is_active=False)
  → access token (15 min) + refresh token (7 days) issued

POST /api/auth/refresh/
  → new access token issued

POST /api/auth/logout/
  → refresh token jti stored in Redis
  → token rejected on all future requests
```

---

The three things that make this production-solid are: `is_active=False` on register so Django blocks login automatically, OTPs in Redis so they auto-expire without any cron jobs, and `jti` blacklisting so logout actually works with stateless JWT.
